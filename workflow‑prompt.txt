/**
 * Detect current Git branch + stage
 * Usage: node gitStatus.js
 */

const { execSync } = require("child_process");

try {
  const branch = execSync("git rev-parse --abbrev-ref HEAD")
    .toString().trim();
  console.log("Git Branch:", branch);

  const stageType = branch.startsWith("feature/")
    ? "Feature"
    : branch.startsWith("fix/")
    ? "Fix"
    : branch === "develop"
    ? "Develop"
    : branch === "main"
    ? "Production"
    : "Unknown";

  console.log("Stage:", stageType);
} catch (error) {
  console.error("Error detecting Git branch", error);
}

2️⃣ /scripts/updateWorkflowStatus.js
const fs = require("fs");
const { execSync } = require("child_process");

try {
  const branch = execSync("git rev-parse --abbrev-ref HEAD").toString().trim();
  const lastCommit = execSync("git rev-parse HEAD").toString().trim();
  
  const statusObj = {
    branch,
    stage: branch.startsWith("feature/")
      ? "feature"
      : branch.startsWith("fix/")
      ? "fix"
      : branch === "develop"
      ? "staging"
      : branch === "main"
      ? "production"
      : "unknown"
  };

  const content = `
stage: ${statusObj.stage}
branch: ${statusObj.branch}
last_commit: ${lastCommit}
  `;
  
  fs.writeFileSync(".workflow-status", content.trim());
  console.log("Workflow status updated!");
} catch (error) {
  console.error("Failed to update status:", error);
}
